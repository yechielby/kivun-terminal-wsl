# =================================================================
# BIDI WRAPPER (Hebrew rendering for Claude Code output)
# =================================================================
# When set to "on" (the default), Claude Code output is piped through
# the kivun-claude-bidi wrapper, which injects Unicode RLE/PDF bracket
# pairs (U+202B / U+202C) around Hebrew runs so they render
# right-to-left regardless of the Konsole profile's BiDi settings.
#
# The installer deploys the wrapper automatically; first launch runs a
# one-time `npm install` (~5-15 s) inside WSL to build its native
# dependencies. Subsequent launches are instant.
#
# If you see rendering issues or want to fall back to unwrapped claude,
# change the value below to "off" and relaunch. The launcher logs the
# wrapper state to BASH_LAUNCH_LOG.txt on every launch.
#
# Default: on
KIVUN_BIDI_WRAPPER=on

# =================================================================
# STRIP BULLET on Hebrew lines (Konsole 23.x bullet-line workaround)
# =================================================================
# When set to "on", the BiDi wrapper removes the leading bullet (●)
# from any line whose first strong char is Hebrew. The bullet alone
# is what was making Konsole 23.x leave Hebrew bullet-lines stuck on
# the LEFT side of the screen: the BiDi engine treats ● as a
# direction-anchoring neutral and won't flip the line RTL. Without
# the bullet, the line's first visible char is Hebrew, BiDi flips
# the line RTL automatically, and the Hebrew renders right-aligned.
#
# Trade-off: you lose the visible bullet marker on Hebrew bullet
# lines (the indentation stays). Most users find this preferable to
# the broken left-alignment. English lines are not affected.
#
# Default: on (user-confirmed fix in v1.1.8 for the Konsole 23.x
# bullet-LTR bug). Set to "off" if you're on Konsole 24.04+ and want
# the visible bullet markers back; that version renders correctly
# without the workaround.
KIVUN_BIDI_STRIP_BULLET=on

# =================================================================
# STRIP INCOMING bidi controls from Claude's stream (v1.1.9)
# =================================================================
# Strips explicit Unicode directional controls — LRE/RLE/PDF/LRO/RLO
# (U+202A-U+202E) and LRI/RLI/FSI/PDI (U+2066-U+2069) — from Claude's
# output before the wrapper processes it, so the wrapper is the only
# thing emitting directional markers. LRM (U+200E) and RLM (U+200F)
# pass through untouched.
#
# Modes:
#   off  - passthrough; Claude's bidi controls reach the terminal as-is
#   auto - strip and count; log a single line on first detection to
#          ~/.local/state/kivun-terminal/bidi-strip.log (default)
#   on   - strip and count; log every chunk where stripping happened
#          (verbose; useful if you're investigating a specific render bug)
#
# Why this is "auto" by default: most observed Claude output contains
# zero directional controls, so the strip is a no-op in practice. The
# value of leaving it on is the side log — a single line per session
# the first time pollution is seen, which lets us tell "Claude polluted
# the stream" apart from "Konsole rendered it wrong" without a full
# packet trace.
KIVUN_BIDI_STRIP_INCOMING=auto

# =================================================================
# DUMP RAW upstream bytes for debugging (v1.1.10)
# =================================================================
# Off by default. Debug-only — turn this on when you have a render
# bug you suspect involves stream pollution but the strip log shows
# zero detections (or you want to see the full byte context around
# what was stripped).
#
# When set to "on", the wrapper appends every chunk Claude sends
# to a side file BEFORE the strip-incoming pass touches it:
#
#     ~/.local/state/kivun-terminal/bidi-raw-dump.bin
#
# The file is appended-to across writes and gets a "session start"
# / "session end" marker line per Kivun launch. To prevent
# unbounded growth, when an existing dump file is larger than
# 5 MiB at session start it gets renamed to .bin.old and a fresh
# .bin file starts. So at any time you have at most the current
# session's dump plus the previous session's rollover.
#
# Performance: writing every chunk to disk adds ~0 ms per chunk
# in normal use because the file is small and Linux page cache
# absorbs writes. Don't leave this on indefinitely though — it
# fills disk over time even with the rotation cap.
#
# To inspect what was on the wire (after running a Kivun session
# with this on):
#
#     # full dump (warning: contains ANSI escapes — use less -R):
#     less -R ~/.local/state/kivun-terminal/bidi-raw-dump.bin
#
#     # just the bidi control chars and surrounding context:
#     grep -aPo '.{0,20}[\x{202A}-\x{202E}\x{2066}-\x{2069}].{0,20}' \
#         ~/.local/state/kivun-terminal/bidi-raw-dump.bin
#
#     # hex view of the bidi bytes (RLE = e2 80 ab, etc.):
#     xxd ~/.local/state/kivun-terminal/bidi-raw-dump.bin | head -40
#
# Default: off
KIVUN_BIDI_DUMP_RAW=off

# =================================================================
# FLATTEN COLORS on RTL lines (v1.1.10) - the actual mixed-content fix
# =================================================================
# Empirically confirmed via Konsole 23.08.5 A/B test (April 2026):
# Konsole's BiDi engine only operates on continuous-attribute regions.
# Any color change inside a Hebrew sentence splits the BiDi run, and
# Qt's text layout positions the resulting fragments incorrectly --
# typically pushing English/code runs to the visual LEFT instead of
# their correct UAX #9 logical position inside the sentence.
#
# This is also documented at the upstream Terminal Working Group:
#   https://terminal-wg.pages.freedesktop.org/bidi/prior-work/terminals.html
# > "Applies BiDi on continuous runs of identical attributes. Any
# > change in e.g. color [...] stops and starts it anew, often
# > resulting in a confusing and incorrect visual behavior."
#
# Stripping SGR escapes from RTL lines makes the whole line one
# attribute run -- Konsole's BiDi gets a clean line to work with and
# LTR runs (English, code, numbers) land in their correct positions.
#
# Modes:
#   off - passthrough; SGR codes reach Konsole as-is on every line
#         (positioning will be broken on mixed Hebrew lines)
#   on  - strip SGR (CSI...m) sequences from any line whose first
#         strong char is Hebrew; LTR lines unaffected (default)
#
# Trade-off: visible loss of syntax color on Hebrew lines. If your
# workflow is mostly English code with occasional Hebrew, you may
# prefer color over positioning -- set this to "off". For users
# whose Claude conversation is mostly Hebrew, "on" is the right
# default (positioning correctness > color).
#
# Important: this strips ONLY SGR sequences (CSI sequences ending
# in 'm'). Cursor positioning, screen clear, OSC window-title, etc.
# all pass through unchanged. LTR lines are NEVER touched.
#
# Default: on
KIVUN_BIDI_FLATTEN_COLORS_RTL=on

# =================================================================
# SKIP PER-RUN RLE/PDF BRACKETING on RTL lines (v1.1.11)
# =================================================================
# THE actual fix for "English/code shifts to the wrong column inside
# Hebrew sentences." Confirmed via Konsole 23.08.5 A/B test (April
# 2026): even with v1.1.10 FLATTEN_COLORS_RTL stripping all SGR,
# real Claude output still mispositioned 'React 19', 'Claude Code',
# numbers, etc. Investigation showed the wrapper's own RLE/PDF
# bracketing (one pair PER Hebrew run on a line) was creating
# multiple attribute-region boundaries that Konsole's per-region
# BiDi mispositions -- exactly what FLATTEN was supposed to fix.
#
# The same lines rendered with NO per-run brackets (just RLM at
# line-start + Konsole's native UAX #9) position correctly. So
# the wrapper now skips per-run bracketing on RTL lines. Hebrew
# runs INSIDE LTR paragraphs (Hebrew embedded in English) still
# get bracketed since the direction marker IS needed there.
#
# Modes:
#   off - no per-run brackets on RTL lines (default v1.1.11+)
#         The line is one attribute region; Konsole positions LTR
#         runs at their UAX #9 logical positions.
#   on  - bracket every Hebrew run regardless of paragraph direction
#         (legacy v1.1.0 - v1.1.10 behavior; reverts to the
#         multiple-attribute-region pattern that misposes LTR runs)
#
# Default: off
KIVUN_BIDI_BRACKET_RTL_RUNS=off
