# Kivun Terminal Configuration
#
# Edit this file to change Kivun's behavior. The top of the file has
# the settings most users tweak; the bottom has BiDi-wrapper internals
# (defaults are good) and the full 23-language reference.


# =================================================================
# QUICK SETTINGS (the things most users want to change)
# =================================================================

# Default flags appended to every `claude` invocation. Mix and match
# as you would type them on the command line — no quoting needed.
#
# Useful Claude Code flags:
#   --continue              Resume your last conversation
#   --resume my-session     Resume a named session
#   --model opus            Use Opus (strongest reasoning)
#   --model sonnet          Use Sonnet (faster, cheaper)
#   --model haiku           Use Haiku (fastest, lightest)
#   --print                 One-off answer, no interactive chat
#   --add-dir C:\full\path  Give Claude access to an extra folder
#   --enable-auto-mode      Skip permission prompts (auto-approve)
#
# Combine: CLAUDE_FLAGS=--continue --model opus --enable-auto-mode
# Default: empty (no extra flags — standard interactive Claude Code)
CLAUDE_FLAGS=

# When true, the desktop shortcut pops a folder picker (browse the
# tree or paste a path) before Konsole opens. When false, the
# shortcut always opens in your Windows home directory.
# Right-click "Open with Kivun Terminal" ignores this setting.
# Default: true
FOLDER_PICKER=true

# Language Claude responds in. Default: english.
# Full list of 23 supported languages at the bottom of this file.
RESPONSE_LANGUAGE=english

# Your primary RTL keyboard layout — used for the Alt+Shift toggle
# inside Konsole. Default: hebrew.
PRIMARY_LANGUAGE=hebrew


# =================================================================
# DISPLAY & INSTALL (occasional changes)
# =================================================================

# Use VcXsrv (true) instead of WSLg (false) for Konsole display.
# VcXsrv enables Alt+Shift keyboard switching; WSLg doesn't.
# Requires VcXsrv installed: https://sourceforge.net/projects/vcxsrv/
# Default: true
USE_VCXSRV=true

# Input text alignment in the terminal:
#   rtl  - Right-aligned (Hebrew/Arabic hugs the right edge)
#   ltr  - Left-aligned  (default terminal behavior)
# Default: rtl
TEXT_DIRECTION=rtl

# How the launcher handles missing Claude Code in WSL:
#   yes  - Auto-install (default)
#   ask  - Prompt [Y/N] each time
#   no   - Skip and show manual install instructions
AUTO_INSTALL_CLAUDE=yes


# =================================================================
# BIDI WRAPPER — Hebrew rendering tunables (advanced; leave alone)
# =================================================================
# The Node.js wrapper pipes Claude Code's output through Unicode
# RLE/PDF/RLM markers so Hebrew renders right-to-left in Konsole.
# Defaults work for most users. Full design notes in
# docs/specs/BIDI_ALGORITHM.md if you need to tune.

# Master switch for the wrapper. Default: on
KIVUN_BIDI_WRAPPER=on

# Strip leading bullet (●) from Hebrew lines (Konsole 23.x bullet-
# direction workaround). Default: on
KIVUN_BIDI_STRIP_BULLET=on

# Strip explicit Unicode directional controls from Claude's stream
# (LRE/RLE/PDF/LRO/RLO/LRI/RLI/FSI/PDI; preserves LRM/RLM).
# Modes: off | auto (single side-log entry on first detect) | on (verbose)
# Default: auto
KIVUN_BIDI_STRIP_INCOMING=auto

# Strip ANSI color codes from RTL lines so Konsole's BiDi gets one
# continuous attribute run per line. Trade-off: no syntax color on
# Hebrew lines. Default: on
KIVUN_BIDI_FLATTEN_COLORS_RTL=on

# Bracket every Hebrew run with RLE/PDF on already-RTL lines. Off is
# correct for Konsole 23.x; on reverts to legacy v1.1.0–v1.1.10
# behavior. Default: off
KIVUN_BIDI_BRACKET_RTL_RUNS=off

# Debug-only: dump raw upstream bytes to
# ~/.local/state/kivun-terminal/bidi-raw-dump.bin (rotated at 5 MiB).
# Useful when investigating wrapper bugs. Default: off
KIVUN_BIDI_DUMP_RAW=off
